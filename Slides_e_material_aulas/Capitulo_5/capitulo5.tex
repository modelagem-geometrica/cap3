\section{Capítulo 5: Variedades Definidas Implicitamente}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 5.0 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Capítulo 5}}

\begin{block}{\bf Capítulo 5: Variedades Definidas Implicitamente}
Neste capítulo são apresentadas técnicas de poligonalização de variedades definidas implicitamente utilizando aproximações simpliciais e sua visualização através de projeções e cortes.
\end{block}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Aproximações Simpliciais}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 5.1 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Aproximações Simpliciais}}


\begin{block}{\bf Aproximações Simpliciais}
Aproximações simpliciais são utilizadas para a poligonalização de uma variedade definida implicitamente. Para isto é necessário obter uma triangulação do domínio da função.
\end{block}

\begin{figure}[h]
\begin{center} 
{\includegraphics[angle=0,scale=0.65]{imagens/triangulacao_j1.pdf}}
\caption{Triangulação $J_1$.} 
\label{fig.j1}
\end{center}
\end{figure}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Marching Simplex}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 5.2 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Marching Simplex}}


\begin{block}{\bf Marching Simplex}
O método marching simplex consiste em percorrer todos os simplexos da triangulação para aproximar a variedade em cada simplexo percorrido.
\end{block}

\begin{figure}[h]
\begin{center} 
{\includegraphics[angle=0,scale=0.20]{imagens/boolean_neptune.jpg}}
\qquad 
{\includegraphics[angle=0,scale=0.20]{imagens/Lucy.jpg}}
\caption{Netuno à esquerda e estátua Lucy à direita.} 
\label{fig.netuno_lucy}
\end{center}
\end{figure}


\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Continuation Simplex}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 5.3 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Continuation Simplex}}

\begin{block}{\bf Continuation Simplex}
O método continuation simplex consiste em percorrer apenas os simplexos da triangulação que são transversais a variedade.
\end{block}

\begin{center}

\end{figure}

\end{frame}

\subsection{Reconstrução de Superfícies a Partir de Nuvens de Pontos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 5.5 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Reconstrução de Superfícies a Partir de Nuvens de Pontos}}

\begin{block}{\bf Reconstrução de Superfícies a Partir de Nuvens de Pontos}
Uma aplicação é a utilização de aproximação RBF (Funções de Base Radial) para reconstruir superfícies a partir de nuvens de pontos que juntamente com o código Marching Simplex será utilizado para gerar uma aproximação da superfície linear por partes.
\end{block}

\begin{figure}[h]
\begin{center} 
{\includegraphics[angle=0,scale=0.40]{imagens/RBF.png}}
\caption{Reconstrução utilizando RBF.} 
\label{fig.RBF}
\end{center}
\end{figure}

\end{frame}
