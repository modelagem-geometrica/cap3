\section{Capítulo 3: Estrutura de Dados para Variedades Lineares por Partes}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 3.0 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Capítulo 3}}

\begin{block}{\bf Capítulo 3: Estrutura de Dados para Variedades Lineares por Partes}
Neste capítulo é introduzido formas de representação e estrutura de dados para Variedades de dimensão 2 e 3, lineares por partes que serão utilizados em capítulos posteriores.
\end{block}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 3.0 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Estrutura de Dados para Variedades Lineares por Partes}}

\begin{block}{\bf Estrutura de Dados para Variedades Lineares por Partes}
Estruturas de dados para variedades de dimensão 2 e 3 lineares por partes são grafos onde temos informações sobre vértices, arestas, faces (e volumes caso seja malhas volumétricas). Estas estruturas podem ser classificadas como estruturas topológicas explícitas ou implícitas.
\end{block}

\begin{figure}[h]
\begin{center} 
{\includegraphics[angle=0,scale=0.34]{imagens/fig410.png}}
\caption{Estrutura de dados topológicas.} 
\label{fig.Est_Dados}
\end{center}
\end{figure}
\end{frame}

\subsection{Estruturas de Dados Explícitas}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 3.1 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Estrutura de Dados Explícitas}}

\begin{block}{\bf Estrutura de Dados Explícitas}
Estruturas de dados topológicas explícitas são estrururas que apresentam todos os elementos da estrutura explicitamente. 
\end{block}

\begin{figure}[h]
\begin{center} 
{\includegraphics[angle=0,scale=0.28]{imagens/fig32.png}}
\caption{Estrutura de dados topológicas explícitas.} 
\label{fig.Explicita}
\end{center}
\end{figure}

\end{frame}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Estruturas de Dados Implícitas}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLIDE 3.2 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{{\bf \color{blue} Estrutura de Dados Implícitas}}

\begin{block}{\bf Estrutura de Dados Implícitas}
Estruturas de dados topológicas implícitas são estrururas que apresentam todos os elementos da estrutura de forma implícita, tendo as informação obtidas através de cálculos com números inteiros. 
\end{block}

\begin{figure}[h]
\begin{center} 
{\includegraphics[angle=0,scale=0.25]{imagens/fig35.png}}
\caption{Estrutura de dados topológicas implícitas.} 
\label{fig.Implicita}
\end{center}
\end{figure}

\end{frame}
